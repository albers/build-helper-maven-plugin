1   package org.codehaus.mojo.buildhelper;
2   
3   import org.apache.maven.execution.MavenSession;
4   import org.apache.maven.plugins.annotations.LifecyclePhase;
5   import org.apache.maven.plugins.annotations.Mojo;
6   import org.apache.maven.plugins.annotations.Parameter;
7   
8   /**
9    * This goal will get the location of the root folder within a multi module build as a property {@code rootlocation}.
10   *
11   * @author Karl Heinz Marbaise
12   * @since 3.0.0
13   */
14  @Mojo( name = "rootlocation", defaultPhase = LifecyclePhase.VALIDATE, threadSafe = true )
15  public class RootLocationMojo
16      extends AbstractDefinePropertyMojo
17  {
18  
19      @Parameter( defaultValue = "${session}", readonly = true, required = true )
20      private MavenSession session;
21  
22      /**
23       * This will cause the execution to be run only at the top of a given module tree.
24       */
25      @Parameter( property = "buildhelper.runOnlyAtExecutionRoot", defaultValue = "false" )
26      private boolean runOnlyAtExecutionRoot;
27  
28      /**
29       * The name of the property in which to store the root location.
30       */
31      @Parameter( defaultValue = "rootlocation" )
32      private String rootLocationProperty;
33  
34      /**
35       * Main plugin execution
36       */
37      public void execute()
38      {
39          if ( runOnlyAtExecutionRoot && !getProject().isExecutionRoot() )
40          {
41              getLog().info( "Skip getting the rootlocation in this project because it's not the Execution Root" );
42          }
43          else
44          {
45              defineProperty( rootLocationProperty, session.getTopLevelProject().getBasedir().getAbsolutePath() );
46          }
47      }
48  
49  }
